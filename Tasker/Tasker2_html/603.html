95             return 0;
96         }
97         if (taskerFolder.mkdir(newDir)) {
98         } else {
99             return -1;
100         }
101         QFile newFile{ taskerFolder.absoluteFilePath(userName + TASKER_FILE_EXTENSION) };
102         if (!newFile.open(QIODevice::WriteOnly)) {
103             return -1;
104         }
105         newFile.close();
106         newFile.flush();
