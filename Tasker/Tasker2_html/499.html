55     /**
56      * @brief CreateCommitmentQWidget::createCommitmentButtonSlot
57      */
58     void CreateCommitmentQWidget::createCommitmentButtonSlot() {
59         this->hide();
60         QVector<Session> sessions{};
61         QVector<Engine::Listener::ListenerType> listeners;
62         if (this->ui->checkKeyboardQCheckBox->checkState() == Qt::Checked) {
63             listeners.push_back(Engine::Listener::ListenerType::keyboard);
64         }
65         if (this->ui->checkMicQCheckBox->checkState() == Qt::Checked) {
66             listeners.push_back(Engine::Listener::ListenerType::audio);
67         }
68         Task newTask{ ui->dropDownTaskQComboBox->currentText(), listeners };
69         sessions.push_back(Session{ newTask });
70         Commitment temp{ this->getCommitmentName(), this->getStartDate(),
71                          this->getEndDate(), this->getInterval(), sessions };
72         udata::User::getInstance()->addCommitment(temp);
73         CommStatsQWidget *wc = MainUI::getInstance()->getCommStats();
74         wc->show();
75     }

109     /**
110      * @brief CreateCommitmentQWidget::getInterval
111      * @return
112      */
113     util::Interval CreateCommitmentQWidget::getInterval() {
114         util::Interval interval{};
115         long long int size, frequency;
116         QString currentSizeLabel = ui->dropDownUnitsLeftQComboBox->currentText();
117         QString currentFrequencyLabel = ui->dropDownUnitsRightQComboBox->currentText();
118         if (currentSizeLabel == MINUTES_STRING) {
119             size = SECONDS_IN_MINUTE * ui->qtyQLineEdit->text().toInt();
120         } else if (currentSizeLabel == HOURS_STRING) {
121             size = SECONDS_IN_HOUR * ui->qtyQLineEdit->text().toInt();
122         }
123         if (currentFrequencyLabel == DAY_STRING) {
124             frequency = SECONDS_IN_DAY;
125         } else if (currentFrequencyLabel == WEEK_STRING) {
126             frequency = SECONDS_IN_WEEK;
127         } else if (currentFrequencyLabel == MONTH_STRING) {
128             frequency = SECONDS_IN_30_DAYS;
129         }
130         interval.size = size;
131         interval.frequency = frequency;
132         return interval;
133     }

144     /**
145      * @brief CreateCommitmentQWidget::dropDownTaskSlot
146      * @param arg1
147      */
148     void CreateCommitmentQWidget::dropDownTaskSlot(const QString &arg1) {
149         if (arg1 == QString(WRITING_STRING)) {
150             this->getKeyboardCheckBox()->setCheckState(Qt::CheckState{ Qt::Checked });
151             this->getAudioCheckBox()->setCheckState(Qt::CheckState{ Qt::Unchecked });
152         } else if (arg1 == QString(MUSIC_STRING)) {
153             this->getAudioCheckBox()->setCheckState(Qt::CheckState{ Qt::Checked });
154             this->getKeyboardCheckBox()->setCheckState(Qt::CheckState{ Qt::Unchecked });
155         } else if (arg1 == QString(CUSTOM_STRING)) {
156             this->getAudioCheckBox()->setCheckState(Qt::CheckState{ Qt::Unchecked });
157             this->getKeyboardCheckBox()->setCheckState(Qt::CheckState{ Qt::Unchecked });
158         }
159     }
