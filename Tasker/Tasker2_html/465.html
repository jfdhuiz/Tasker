38     /**
39      * @brief AudioMachine::AudioMachine
40      */
41     AudioMachine::AudioMachine() : audioDevice(nullptr), qAudioInput(nullptr) {
42         QAudioFormat format;
43     
44         // Set up the desired format, for example:
45         format.setSampleRate(8000);
46         format.setChannelCount(1);
47         format.setSampleSize(8);
48         format.setCodec("audio/pcm");
49         format.setByteOrder(QAudioFormat::LittleEndian);
50         format.setSampleType(QAudioFormat::UnSignedInt);
51     
52         // destinationFile.setFileName("/tmp/test.raw");
53         // destinationFile.open( QIODevice::WriteOnly | QIODevice::Truncate );
54     
55         audioDevice = new AudioDevice(format);
56         audioDevice->open(QIODevice::WriteOnly);
57     
58         // assign audio device here
59         QAudioDeviceInfo info = QAudioDeviceInfo::defaultInputDevice();
60     
61         if (!info.isFormatSupported(format)) {
62             qWarning()
63                 << "Default format not supported, trying to use the nearest.";
64             format = info.nearestFormat(format);
65         }
66     
67         qAudioInput = new QAudioInput(format, this);
68         // connect(audio, SIGNAL(stateChanged(QAudio::State)), this, SLOT(handleStateChanged(QAudio::State)));
69     
70         // QTimer::singleShot(3000, this, SLOT(stopRecording()));
71         // qDebug()<<"calling start on destinationFile ";
72         // qDebug()<<"audiomachine thread:"<<QThread::currentThreadId();
73     
74         // show device name on console
75         qDebug() << info.deviceName();
76     
77         qAudioInput->start(audioDevice);
78     
79         qAudioInput->setVolume(0.0);
80         audioDevice->setMinAmplitude(audioDevice->getDeviceLevel());
81         qAudioInput->setVolume(1.0);
82     
83         // Records audio for 3000ms
84     }

110     /**
111      * @brief AudioMachine::handleStateChanged
112      * @param newState
113      */
114     void AudioMachine::handleStateChanged(QAudio::State newState) {
115         switch (newState) {
116         case QAudio::StoppedState:
117             if (qAudioInput->error() != QAudio::NoError) {
118                 // Error handling
119             } else {
120                 // Finished recording
121                 qDebug() << "stopped state";
122             }
123     
124             break;
125     
126         case QAudio::ActiveState:
127             // Started recording - read from IO device
128             qDebug() << "active state";
129             break;
130     
131         default:
132             // ... other cases as appropriate
133             break;
134         }
135     }

137     /**
138      * @brief AudioMachine::stopRecording
139      */
140     void AudioMachine::stopRecording() {
141         qAudioInput->stop();
142         audioDevice->close();
143     
144         delete qAudioInput;
145         qAudioInput = nullptr;
146     
147         delete audioDevice;
148         audioDevice = nullptr;
149     }
