75     /**
76      * @brief udata::operator <<This writes a commitment(and all of its fields) to a data stream(a file).
77      * @param out The data stream to write to.
78      * @param newCommitment The in-memory commitment to write to data stream.
79      * @return The data stream. Very useful for error checking.
80      */
81     QDataStream &udata::operator<<(QDataStream &out, const udata::Commitment &newCommitment) {
82         unsigned long long int size, frequency;
83         size = newCommitment.interval.size;
84         frequency = newCommitment.interval.frequency;
85         out << newCommitment.name << newCommitment.dateStart << newCommitment.dateEnd
86             << newCommitment.interval << newCommitment.sessions;
87         return out;
88     }

90     /**
91      * @brief udata::operator >> Writes all data from data stream into newCommitment.
92      * @param in data stream to read from, a file.
93      * @param newCommitment The in-memory commitment to be written to.
94      * @return The data stream out. Very useful for error-checking.
95      */
96     QDataStream &udata::operator>>(QDataStream &in, udata::Commitment &newCommitment) {
97         util::Interval commitmentInterval;
98         QString commitmentName;
99         QDate commitmentDateStart;
100         QDate commitmentDateEnd;
101         QVector<Session> commitmentSessions;
102         in >> commitmentName >> commitmentDateStart >> commitmentDateEnd >>
103             commitmentInterval >> commitmentSessions;
104         newCommitment.name = commitmentName;
105         newCommitment.dateStart = commitmentDateStart;
106         newCommitment.dateEnd = commitmentDateEnd;
107         newCommitment.interval = commitmentInterval;
108         newCommitment.sessions = commitmentSessions;
109         return in;
110     }
