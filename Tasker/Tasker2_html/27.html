64     qint64  QAudioDevice::writeData(const char *data, qint64 len)
65     {
66     //    qDebug()<<"writeData begin";
67     //    qDebug()<<"writeData end";
68         if (maxAmplitude) {
69             Q_ASSERT(format.sampleSize() % 8 == 0);
70             const int channelBytes = format.sampleSize() / 8;
71             const int sampleBytes = format.channelCount() * channelBytes;
72             Q_ASSERT(len % sampleBytes == 0);
73             const int numSamples = len / sampleBytes;
74     
75             quint32 maxValue = 0;
76             const unsigned char *ptr = reinterpret_cast<const unsigned char *>(data);
77     
78             for (int i = 0; i < numSamples; ++i) {
79                 for (int j = 0; j < format.channelCount(); ++j) {
80                     quint32 value = 0;
81     
82                     if (format.sampleSize() == 8 && format.sampleType() == QAudioFormat::UnSignedInt) {
83                         value = *reinterpret_cast<const quint8*>(ptr);
84                     } else if (format.sampleSize() == 8 && format.sampleType() == QAudioFormat::SignedInt) {
85                         value = qAbs(*reinterpret_cast<const qint8*>(ptr));
86                     } else if (format.sampleSize() == 16 && format.sampleType() == QAudioFormat::UnSignedInt) {
87                         if (format.byteOrder() == QAudioFormat::LittleEndian)
88                             value = qFromLittleEndian<quint16>(ptr);
89                         else
90                             value = qFromBigEndian<quint16>(ptr);
91                     } else if (format.sampleSize() == 16 && format.sampleType() == QAudioFormat::SignedInt) {
92                         if (format.byteOrder() == QAudioFormat::LittleEndian)
93                             value = qAbs(qFromLittleEndian<qint16>(ptr));
94                         else
95                             value = qAbs(qFromBigEndian<qint16>(ptr));
96                     } else if (format.sampleSize() == 32 && format.sampleType() == QAudioFormat::UnSignedInt) {
97                         if (format.byteOrder() == QAudioFormat::LittleEndian)
98                             value = qFromLittleEndian<quint32>(ptr);
99                         else
100                             value = qFromBigEndian<quint32>(ptr);
101                     } else if (format.sampleSize() == 32 && format.sampleType() == QAudioFormat::SignedInt) {
102                         if (format.byteOrder() == QAudioFormat::LittleEndian)
103                             value = qAbs(qFromLittleEndian<qint32>(ptr));
104                         else
105                             value = qAbs(qFromBigEndian<qint32>(ptr));
106                     } else if (format.sampleSize() == 32 && format.sampleType() == QAudioFormat::Float) {
107                         value = qAbs(*reinterpret_cast<const float*>(ptr) * 0x7fffffff); // assumes 0-1.0
108                     }
109     
110                     maxValue = qMax(value, maxValue);
111                     ptr += channelBytes;
112                 }
113             }
114     
115             maxValue = getMin(maxValue, maxAmplitude);
116             deviceLevel = qreal(maxValue) / maxAmplitude;
117     //        qDebug()<<"device level:"<< deviceLevel;
118         }
119     //    qDebug()<<"thread id on write:"<<QThread::currentThreadId();
120     
121         return len;
122     }
