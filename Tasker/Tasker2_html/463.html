29     #include "AudioThread.h"
30     #include <QDebug>
31     
32     using Engine::AudioMachine;
33     using Engine::AudioThread;
34     
35     /**
36      * @brief AudioThread::AudioThread
37      */
38     AudioThread::AudioThread() : audioMachine(nullptr), audioLevel(nullptr) {
39         connect(&qThread, &QThread::started, this, &AudioThread::updateState);
40         qDebug() << "AudioThread constructor thread id: << " << QThread::currentThreadId();
41         this->moveToThread(&qThread);
42         qThread.start();
43     }
44     
45     /**
46      * @brief AudioThread::~AudioThread
47      */
48     AudioThread::~AudioThread() {
49         delete audioMachine;
50     }
51     
52     /**
53      * @brief AudioThread::getQThread
54      * @return
55      */
56     QThread &AudioThread::getQThread() {
57         return qThread;
58     }
59     
60     /**
61      * @brief AudioThread::getAudioMachine
62      * @return
63      */
64     AudioMachine *&AudioThread::getAudioMachine() {
65         return audioMachine;
66     }
67     
68     /**
69      * @brief AudioThread::getAudioLevel
70      * @return
71      */
72     qreal AudioThread::getAudioLevel() {
73         return audioLevel == nullptr ? 0.0 : (*audioLevel);
74     }
75     
76     /**
77      * @brief AudioThread::updateState
78      */
79     void AudioThread::updateState() {
80         audioMachine = new AudioMachine();
81         qDebug() << "AudioThread updateState() thread id: << " << QThread::currentThreadId();
82         audioLevel = &(audioMachine->getAudioDevice()->getDeviceLevel());
83     }
