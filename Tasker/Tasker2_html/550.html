119     /**
120      * @brief AudioListener::startListening
121      * @param delay
122      * @return
123      */
124     int AudioListener::startListening(unsigned long int delay) {
125         audioThread = new AudioThread;
126         // must pass in audio device type at some point to thread.
127     
128         // connect(address of thread object, &thisThread,
129         // signal want to run, &QThread::finished,
130         // object that contains slot, this
131         // function i want to run, &AudioListener::cleanup);
132     
133         connect(&audioThread->getQThread(), &QThread::finished, this, &AudioListener::cleanup);
134     //    connect();
135         qDebug() << "From startListening on Listener.cpp: " << QThread::currentThreadId();
136         audioThread->moveToThread(&audioThread->getQThread());
137     
138         audioListenerState = AudioListenerState::ON;
139     
140         while (true) {
141             if (audioThread->getAudioMachine()) {
142                 if (audioThread->getAudioMachine()->getAudioDevice()) {
143                     audioThread->getAudioMachine()->getQAudioInput()->setVolume(0.0);
144                     while (audioThread->getAudioMachine()->getAudioDevice()->getDeviceLevel() == 0.0) {
145                     }
146                     audioThread->getAudioMachine()->getAudioDevice()->setMinAmplitude(
147                         audioThread->getAudioMachine()->getAudioDevice()->getDeviceLevel());
148                     audioThread->getAudioMachine()->getQAudioInput()->setVolume(1.0);
149                     break;
150                 }
151             }
152         }
153         qDebug() << "AudioListener updateState() thread id: << " << QThread::currentThreadId();
154         while (true) {
155             ListenerState state;
156     
157             state = audioThread->getAudioLevel() > audioThreshold ?
158                         ListenerState::productive :
159                         ListenerState::unproductive;
160             setState(state);
161     
162     //        qDebug() << "listener level: " << audioThread->getAudioLevel();
163     
164             if (state == ListenerState::productive) {
165     //            qDebug() << "status: productive";
166             } else {
167     //            qDebug() << "status: unproductive";
168             }
169     
170             QThread::sleep(BASE_DELAY + delay);
171         }
172     
173         return EXIT_SUCCESS;
174     }
